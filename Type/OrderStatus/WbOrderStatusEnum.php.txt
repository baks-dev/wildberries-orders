namespace BaksDev\Wildberries\Orders\Type\OrderStatus;

enum WbOrderStatusEnum : int
{
	/** Новое сборочное задание */
	case NEW = 0;
	
	/** На сборке */
	case ACCEPTED = 1;
	
	/** В доставке  */
	case COMPLETE = 2;
	
	/** Отменено продавцом */
	case CANCEL = 3;
	
	/** сборочное задание отсортировано */
	case SORTED = 4;
	
	/** сборочное задание получено покупателем */
	case SOLD = 5;
	
	/**  отмена сборочного задания */
	case CANCELED = 6;
	
	/**  отмена сборочного задания покупателем */
	case CLIENT = 7;
	
	
	public static function fromOrderStatus($status) : WbOrderStatus
	{
		return match ($status)
		{
			'new' => new WbOrderStatus(self::NEW),
			'confirm' => new WbOrderStatus(self::ACCEPTED),
			'complete' => new WbOrderStatus(self::COMPLETE),
			'cancel' => new WbOrderStatus(self::CANCEL),
			
			'sorted' => new WbOrderStatus(self::SORTED),
			'sold' => new WbOrderStatus(self::SOLD),
			'canceled' => new WbOrderStatus(self::CANCELED),
			'canceled_by_client' => new WbOrderStatus(self::CLIENT)
			
		};
	}
	
}
