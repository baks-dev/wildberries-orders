namespace BaksDev\Wildberries\Orders\Controller\Admin;



use BaksDev\Core\Controller\AbstractController;
use BaksDev\Core\Form\Search\SearchDTO;
use BaksDev\Core\Form\Search\SearchForm;
use BaksDev\Wildberries\Orders\Repository\AllOrders\AllOrdersInterface;
use Sensio\Bundle\FrameworkExtraBundle\Configuration\Security;
use Symfony\Component\HttpFoundation\Request;
use Symfony\Component\HttpFoundation\Response;
use Symfony\Component\Routing\Annotation\Route;

//#[IsGranted(new Expression('"ROLE_ADMIN" in role_names or "ROLE_WB_ORDERS" in role_names'))]
#[Security("is_granted('ROLE_ADMIN') or is_granted('ROLE_WB_ORDERS')")]
final class IndexController extends AbstractController
{
    /** Список всех заказов Wildberries */
    #[Route('/admin/wb/orders/{page<\d+>}', name: 'admin.index', methods: ['GET', 'POST'])]
    public function index(
      Request $request,
      AllOrdersInterface $orders,
      int $page = 0,
    ) : Response
    {
        /* Поиск */
        $search = new SearchDTO();
        $searchForm = $this->createForm(SearchForm::class, $search);
        $searchForm->handleRequest($request);
        
        /* Получаем список */
        $stmt = $orders->get($search);
        $query = new Paginator($page, $stmt, $request);
        
        return $this->render(
          [
            'query' => $query,
            //'counter' => $getAllProduct->count(),
            'search' => $searchForm->createView(),
          ]);
    }
    
}
